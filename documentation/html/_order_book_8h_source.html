23  template<eBookSide BOOKSIDE, eBookSide CROSSBOOKSIDE>
24  void add(const UniqueOrderId::type& uniqueOrderId, const BookOrder& order)
25  {
26  outputCallback(Acknowledgment(order.userId, order.id));
27  if (order.price == BookOrder::PRICE_SPECIAL_MEANING_MARKET_ORDER)
28  {
29  if (isAny<CROSSBOOKSIDE>())
30  {
31  doMarketOrderMatches<BOOKSIDE, CROSSBOOKSIDE>(order);
32  }
33  else
34  {
35  // There is no orders in the order book to match the market order. Assume canceled
36  }
37  }
38  else // order type is Limit
39  {
40  if (isAny<CROSSBOOKSIDE>() && isBestPriceCrossed<CROSSBOOKSIDE>(order.price))
41  {
42  doMatches<BOOKSIDE, CROSSBOOKSIDE>(order);
43  }
44  else
45  {
46  insertOrder<BOOKSIDE>(order);
47  }
48  }

51  template<eBookSide BOOKSIDE>
52  void remove(const UniqueOrderId::type& uniqueOrderId)
53  {
54  auto iter = idToOrderMap.find(uniqueOrderId);
55  if (iter == idToOrderMap.end())
56  throw std::runtime_error("The order can't be found in the order book");
57  auto& ord = iter->second;
58  outputCallback(Acknowledgment(ord->userId, ord->id));
59 
60  BookPrice<BOOKSIDE> key(ord->price);
61  BookIterator<BOOKSIDE> bookIterator;
62  BookSide<BOOKSIDE>& bookSide = getBookSide<BOOKSIDE>();
63  for (bookIterator = bookSide.find(key); bookIterator != bookSide.end(); ++bookIterator)
64  {
65  // If this is the correct quote erase it
66  if (bookIterator->second->createUniqueOrderId() == uniqueOrderId)
67  {
68  if (getBestPrice<BOOKSIDE>() == ord->price)
69  {
70  auto topBookQty = getTopBookQty<BOOKSIDE>();
71  if (topBookQty == ord->qty)
72  {
73  eraseOrder<BOOKSIDE>(bookIterator);
74  BookIterator<BOOKSIDE> iter = bookSide.begin();
75  auto newBestPrice = (iter == bookSide.end() ? BookOrder::PRICE_UNDEFINED : iter->first.price);
76  setBestPrice<BOOKSIDE>(newBestPrice);
77  int totalTopBookQty = 0;
78  while (iter != bookSide.end() && iter->first.price == newBestPrice)
79  {
80  totalTopBookQty += iter->second->qty;
81  ++iter;
82  }
83  setTopBookQty<BOOKSIDE>(totalTopBookQty);
84  reportOnTopOfBookChange<BOOKSIDE>();
85  }
86  else
87  {
88  topBookQty -= ord->qty;
89  setTopBookQty<BOOKSIDE>(topBookQty);
90  reportOnTopOfBookChange<BOOKSIDE>();
91  eraseOrder<BOOKSIDE>(bookIterator);
92  }
93 
94  }
95  else
96  {
97  eraseOrder<BOOKSIDE>(bookIterator);
98  }
99  return;
100  }
101  else if (key < bookIterator->first)
102  {
103  // exit early if result is beyond the matching prices
104  throw std::runtime_error("The order can't be found in the order book");
105  }
106  }

311 template<eBookSide BOOKSIDE, eBookSide CROSSBOOKSIDE> inline
312 void OrderBook::doMatches(const BookOrder& ord)
313 {
314  Matcher<CROSSBOOKSIDE> matcher(ord, getBookSide<CROSSBOOKSIDE>().begin(), getBookSide<CROSSBOOKSIDE>().end(),
315  [this](BookIterator<CROSSBOOKSIDE>& it){ eraseOrder<CROSSBOOKSIDE>(it); });
316  while (true) // loop by levels
317  {
318  while (true) // loop within one level
319  {
320  if (matcher.isQuoteMatchingAllowed())
321  {
322  matcher.createTrade(outputCallback);
323  }
324  else
325  {
326  throw std::runtime_error("User can't trade against his own orders");
327  }
328 
329  if (matcher.isNextQuoteOnLevelExist() && matcher.isNextQuoteCrossed() && matcher.getFilledAmount() < ord.qty)
330  matcher.doStepToNextQuote();
331  else
332  break;
333  }
334 
335  if (matcher.isNextQuoteExist() && matcher.isNextQuoteCrossed() && matcher.getFilledAmount() < ord.qty)
336  matcher.doStepToNextQuote();
337  else
338  break;
339  }
340 
341  if (matcher.getFilledAmount() < ord.qty)
342  {
343  // order was partially filled
344  BookOrder orderToInsert = ord;
345  orderToInsert.qty -= matcher.getFilledAmount();
346  insertOrder<BOOKSIDE>(orderToInsert);
347  }
348  auto [newBestPrice, totalTopBookQty] = matcher.getBestPriceAndTotalTopBookQty();
349  setTopBookQty<CROSSBOOKSIDE>(totalTopBookQty);
350  auto bestPrice = getBestPrice<CROSSBOOKSIDE>();
351  if (bestPrice != newBestPrice)
352  {
353  setBestPrice<CROSSBOOKSIDE>(newBestPrice);
354  }
355  reportOnTopOfBookChange<CROSSBOOKSIDE>();
356 }

10 class TestsOrderBooksProcessor
11 {

15  {
16  }
17 
18  std::string feedLinesAndGetOutput(const std::string& lines)
19  {
20  for (const auto &line : InputStringParser::splitLines(lines))
21  {
22  orderBooksContainer.processOrder(InputStringParser::parse(line));
23  }
24  return ss.str();
25  }
26 
27 private:
28  void processOutput(const OutputData& outputData)
29  {
30  ss << outputData.toString() << std::endl;
31  }
